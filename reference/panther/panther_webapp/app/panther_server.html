    def add_header(r):
        """
        It sets the cache control headers to prevent caching

        r.headers["Cache-Control"] = "no-cache, no-store, must-revalidate"

        r.headers["Expires"] = "0"

        r.headers.add("Access-Control-Allow-Origin", "*")
        return r

    def update_count():
